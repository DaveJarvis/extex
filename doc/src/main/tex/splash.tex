\newcommand\SIDE[1]{\begingroup\LARGE%
  \setbox\splashbox=\hbox{#1}%
  \splashwidth=\wd\splashbox    \advance\splashwidth 1em
  \splashheight=\ht\splashbox   \advance\splashheight 1em
  \begin{tikzpicture}[inner sep=.5em]
    \clip[xshift=-\splashheight/2,yshift=-\splashwidth/2]
         (0,0) rectangle(\splashheight,\splashwidth); 
    \fill[xshift=-.5*\splashheight,yshift=-.5*\splashwidth,path fading=Fade]
         (0mm,0mm) rectangle(\splashheight,\splashwidth);
    \foreach \x in {.02mm,.04mm,.06mm,.08mm,.1mm,.12mm,.14mm,.16mm,.18mm}
      \draw (-\x,\x) node[rotate=90,color=white]{\copy\splashbox};
    \foreach \x in {.02mm,.04mm,.06mm,.08mm,.1mm,.12mm,.14mm,.16mm,.18mm}
      \draw (\x,-\x) node[rotate=90,color=white!25!black]{\copy\splashbox};
    \draw (0mm,0mm) node[rotate=90,color=white!80!black]{\copy\splashbox};
  \end{tikzpicture}\endgroup
}
