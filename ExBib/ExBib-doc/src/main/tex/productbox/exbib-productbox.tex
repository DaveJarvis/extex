\def\wave#1#2#3#4#5#6{
  \xx=#1mm
  \foreach \x in {95,90,85,80,75,70,65,60,55,50,45,40,35,30,25,20,15,10} {
    \draw [yshift=#2\the\xx,red,line width=.7mm,draw opacity=.0\x]
    (#3mm,#4mm) .. controls (#5mm,#4mm) and (#5mm,#6mm) .. (0mm,#6mm);
    \global\advance\xx1mm
  }
}
